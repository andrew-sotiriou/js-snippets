Quick Sort

This is an example of Quick Sort.

Example:

//First we add a default Comparator to be used with the Quick Sort
const defaultComparator = (a, b) => {
if (a < b) {
return -1;
}
if (a > b) {
return 1;
}
return 0;
};
//Create an array to sort:
const arrayToSort = [5,52,5,28,1,4,3,5,9,10,467389,998348];
//Now lets recursively sort the array:
const quickSort = (unsortedArray, comparator = defaultComparator) => {
const sortedArray = [ ...unsortedArray ];
const recursiveSort = (start, end) => {
if (end - start < 1) {
return;
}
const pivotValue = sortedArray[end];
let splitIndex = start;
for (let i = start; i < end; i++) {
const sort = comparator(sortedArray[i], pivotValue);
if (sort === -1) {
if (splitIndex !== i) {
const temp = sortedArray[splitIndex];
sortedArray[splitIndex] = sortedArray[i];
sortedArray[i] = temp;
}
splitIndex++;
}
}
sortedArray[end] = sortedArray[splitIndex];
sortedArray[splitIndex] = pivotValue;
recursiveSort(start, splitIndex - 1);
recursiveSort(splitIndex + 1, end);
};
recursiveSort(0, unsortedArray.length - 1);
return sortedArray;
};
quickSort(arrayToSort)

Search

Breadth First Search

This is an example of a Breadth First Search.

Example:

// First we need a tree for the breadth First Search
const tree = [
{value: 6, left: 1, right: 2},
{value: 5, left: 3, right: 4},
{value: 7, left: null, right: 5},
{value: 3, left: 6, right: null},
{value: 4, left: null, right: null},
{value: 9, left: 7, right: 8},
{value: 2, left: 9, right: null},
{value: 8, left: null, right: null},
{value: 10, left: null, right: null},
{value: 1, left: null, right: null}
];
// Now that we have a tree we can run a Breadth First Search.
const breadthFirstSearch = (tree, value) => {
let queue = []
queue.push(tree[0])
while (queue.length !== 0) {
for (let i = 0; i < queue.length; i++) {
let node = queue.shift()
if (node.value === value) {
return node;
}
if (node.left) {
queue.push(tree[node.left]);
}
if (node.right) {
queue.push(tree[node.right]);
}
}
}
return null;
}
breadthFirstSearch(tree, 6);

Depth First Search

This is an example of a Depth First Search.

Example:

// First we need a tree for the breadth First Search
const tree = [
{value: 6, left: 1, right: 2},
{value: 5, left: 3, right: 4},
{value: 7, left: null, right: 5},
{value: 3, left: 6, right: null},
{value: 4, left: null, right: null},
{value: 9, left: 7, right: 8},
{value: 2, left: 9, right: null},
{value: 8, left: null, right: null},
{value: 10, left: null, right: null},
{value: 1, left: null, right: null}
];
// Now that we have a tree we can run a Depth First Search.
const depthFirstSearch = (tree, value) => {
let stack = []
stack.push(tree[0])
while (stack.length !== 0) {
for (let i = 0; i < stack.length; i++) {
let node = stack.pop();
if (node.value === value) {
return node;
}
if (node.right) {
stack.push(tree[node.right]);
}
if (node.left) {
stack.push(tree[node.left]);
}
}
}
return null;
}
depthFirstSearch(tree, 6);
